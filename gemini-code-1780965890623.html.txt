    onlineRoomId = null;
    onlineMyRole = null;
    document.getElementById('blackNickDisplay').textContent = '로컬 유저1';
    document.getElementById('whiteNickDisplay').textContent = '로컬 유저2';

  }
}

    if(playerDecks[role].length > 0) {

    }
  }
}

/* 손패 관리 및 카드 결합(Merge) 로직 구현 */
function renderHands(role) {

  }

      <div class="hand-card mine ${isSel?'selected':''}" onclick="clickHandCard('${role}', ${idx})">

  for(let i=0; i<2; i++) {

function clickHandCard(role, idx) {
  if(!localGame && onlineMyRole !== role) return;
  if(currentKey() !== role) return;
  if(phase !== 'card') return;

  if(selectedHandIndex === idx) {
    selectedHandIndex = null;
  } else {
    selectedHandIndex = idx;
  }
  renderHands(role);
  updateActionButtons();
}

function clickMergeSlot(role, slotIdx) {
  if(!localGame && onlineMyRole !== role) return;
  if(currentKey() !== role) return;
  if(phase !== 'card') return;

  const slot = mergeSlots[role];
  if(slot[slotIdx]) {
    // 슬롯 해제 후 손패 복구
    playerHands[role].push(slot[slotIdx]);
    slot[slotIdx] = null;
    selectedHandIndex = null;
  } else {
    // 슬롯이 비어있고 장착 대상 카드가 선택된 상태일 때
    if(selectedHandIndex !== null) {
      const card = playerHands[role][selectedHandIndex];
      slot[slotIdx] = card;
      playerHands[role].splice(selectedHandIndex, 1);
      selectedHandIndex = null;
    }
  }
  renderHands(role);
  updateActionButtons();
}

// [핵심 변경점] 별 카드 합성 차단 예외 처리 로직 반영 완료
function doMerge(role) {
  const slot = mergeSlots[role];
  if(!slot[0] || !slot[1]) return;

  // 예외 조건 분석: 결합 재료에 하나라도 별 속성이 존재하면 거부
  if(slot[0].type === '별' || slot[1].type === '별') {
    alert("별 카드는 합성 재료로 사용할 수 없습니다!");
    // 안전하게 복구 처리 자동 실행
    playerHands[role].push(slot[0]);
    playerHands[role].push(slot[1]);
    slot[0] = null; slot[1] = null;
    renderHands(role);
    return;
  }

  if(slot[0].star !== slot[1].star) {
    alert("서로 동일한 성급 카드끼지만 합성이 가능합니다.");
    return;
  }

  const targetStar = slot[0].star + 1;
  if(targetStar > 5) {
    alert("최대 성급(5성)에 도달해 더 이상 합성할 수 없습니다.");
    return;
  }

  // 매칭 풀 등급에 부합하는 랜덤 성급 진화 카드 배포
  const pool = CARDS_DEF.filter(c => c.star === targetStar);
  const rewards = pool[Math.floor(Math.random() * pool.length)];
  const evolvedCard = {...rewards, uuid: Math.random().toString(36).substr(2,9)};

  slot[0] = null;
  slot[1] = null;
  playerHands[role].push(evolvedCard);
  
  showCardAnimation(role === 'black' ? '흑' : '백', evolvedCard);
  renderHands(role);
  updateActionButtons();
  
  if(!localGame) synchronizeOnlineState();
}

  const useBtn = document.getElementById(pCode+'UseCardBtn');

  const placeBtn = document.getElementById(pCode+'PlaceBtn');

    passBtn.style.display = 'none';
    placeBtn.style.display = 'none';

  }

  if(selectedHandIndex !== null) {
    useBtn.style.display = 'block';
    passBtn.style.display = 'none';
  } else {
    useBtn.style.display = 'none';
    passBtn.style.display = 'block';
  }
  
  // 착수 가이드 플래그 검사 상태 매핑
  if(phase === 'place') {
    useBtn.style.display = 'none';
    passBtn.style.display = 'none';
    placeBtn.style.display = 'block';
    placeBtn.disabled = (tempPlacement === null);
  } else {
    placeBtn.style.display = 'none';
  }
}

function cancelCardUse() {
  cardUnderUse = null;
  closeModal('effectModal');
}

function confirmCardUse() {
  closeModal('effectModal');
  if(!cardUnderUse) return;
  
  const k = currentKey();
  // 타겟팅 선택 스킬이 아닌 즉발성 강화 드로우 계열 카드 연산 처리
  if(cardUnderUse.effect === 'replace') {
    playerHands[k].splice(selectedHandIndex, 1);
    dealCards(k, 1);
    selectedHandIndex = null;
    finishCardPhase();
  } else if(cardUnderUse.effect === 'draw2') {
    playerHands[k].splice(selectedHandIndex, 1);
    dealCards(k, 2);
    selectedHandIndex = null;
    finishCardPhase();
  } else {
    // 보드 타겟팅 스킬 카드 모드 진입 지동 처리 전환
    setStatus(`<b>[${cardUnderUse.name}]</b> 대상을 보드판 위에서 선택해주세요.`);
    selectedHandIndex = null;
    drawBoard();
  }
}

function executeBoardCardEffect(r, c) {
  const k = currentKey();
  const eff = cardUnderUse.effect;
  const targetP = board[r][c];
  
  if(eff === 'fortify') {
    const myStone = currentPlayer;
    if(targetP !== myStone) { alert("자신의 돌만 강화 지정이 가능합니다."); return; }
    fortifiedStones[`${r},${c}`] = true;
  }
  else if(eff === 'remove') {
    if(targetP === 0) { alert("공백 칸은 파괴 대상이 될 수 없습니다."); return; }
    if(fortifiedStones[`${r},${c}`]) { alert("굳혀진 돌은 파괴 효과에 면역입니다."); return; }
    board[r][c] = 0;
  }
  
  // 패 사용 정산 및 페이즈 종료 동기화
  const cardIdx = playerHands[k].findIndex(c => c.uuid === cardUnderUse.uuid);
  if(cardIdx !== -1) playerHands[k].splice(cardIdx, 1);
  
  showCardAnimation(k === 'black' ? '흑' : '백', cardUnderUse);
  cardUnderUse = null;
  finishCardPhase();
}

function finishCardPhase() {
  renderAll();
  passToPlace();
  if(!localGame) synchronizeOnlineState();
}

/* 캔버스 기반 보드 엔진 및 [착수 취소 가이드라인 알고리즘] 탑재 */
let canvas, ctx;
const CELL_SIZE = 26;

function initBoardCanvas() {
  canvas = document.getElementById('boardCanvas');
  canvas.width = BOARD_SIZE * CELL_SIZE + CELL_SIZE;
  canvas.height = BOARD_SIZE * CELL_SIZE + CELL_SIZE;
  ctx = canvas.getContext('2d');
  
  canvas.removeEventListener('pointerdown', handleBoardClick);
  canvas.addEventListener('pointerdown', handleBoardClick);
}

function drawBoard() {
  if(!ctx) return;
  ctx.clearRect(0, 0, canvas.width, canvas.height);
  
  ctx.strokeStyle = '#333322';
  ctx.lineWidth = 1.2;
  
  // 격자 라인 드로잉 연산
  for(let i=0; i<BOARD_SIZE; i++) {
    const offset = CELL_SIZE + i * CELL_SIZE;
    ctx.beginPath(); ctx.moveTo(CELL_SIZE, offset); ctx.lineTo(BOARD_SIZE * CELL_SIZE, offset); ctx.stroke();
    ctx.beginPath(); ctx.moveTo(offset, CELL_SIZE); ctx.lineTo(offset, BOARD_SIZE * CELL_SIZE); ctx.stroke();
  }
  
  // 화점 위치 지정 마킹
  const stars = [3, 7, 11];
  ctx.fillStyle = '#333322';
  stars.forEach(r => {
    stars.forEach(c => {
      ctx.beginPath();
      ctx.arc(CELL_SIZE + c*CELL_SIZE, CELL_SIZE + r*CELL_SIZE, 3, 0, 2*Math.PI);
      ctx.fill();
    });
  });

  // 보드 전체 노드 돌 스캔 및 드로잉
  for(let r=0; r<BOARD_SIZE; r++) {
    for(let c=0; c<BOARD_SIZE; c++) {
      if(board[r][c] !== 0) {
        drawStone(r, c, board[r][c], fortifiedStones[`${r},${c}`]);
      }
    }
  }

  // [중요] 임시 미확정 가이드라인 드로잉 가젯 세션 파트
  if(phase === 'place' && tempPlacement) {
    drawStone(tempPlacement.r, tempPlacement.c, currentPlayer, false, true);
  }
}

function drawStone(r, c, p, isFort, isTemp=false) {
  const cx = CELL_SIZE + c * CELL_SIZE;
  const cy = CELL_SIZE + r * CELL_SIZE;
  const radius = CELL_SIZE * 0.44;

  ctx.save();
  if(isTemp) ctx.globalAlpha = 0.55; // 미확정 시 반투명 가이드 효과 부여

  ctx.beginPath();
  ctx.arc(cx, cy, radius, 0, 2 * Math.PI);
  
  if(p === 1) {
    const grad = ctx.createRadialGradient(cx - radius*0.15, cy - radius*0.2, radius*0.1, cx, cy, radius);
    grad.addColorStop(0, '#555555'); grad.addColorStop(1, '#111111');
    ctx.fillStyle = grad; ctx.fill();
  } else {
    const grad = ctx.createRadialGradient(cx - radius*0.15, cy - radius*0.2, radius*0.1, cx, cy, radius);
    grad.addColorStop(0, '#ffffff'); grad.addColorStop(1, '#aaaaaa');
    ctx.fillStyle = grad; ctx.fill();
    ctx.strokeStyle = '#999999'; ctx.lineWidth = 1; ctx.stroke();
  }

  // 굳히기(Fortify) 시각 효과 오버레이 드로잉
  if(isFort) {
    ctx.strokeStyle = varColor('--accent');
    ctx.lineWidth = 2.5;
    ctx.beginPath();
    ctx.arc(cx, cy, radius + 2, 0, 2*Math.PI);
    ctx.stroke();
  }
  ctx.restore();
}

function varColor(varName) {
  return getComputedStyle(document.documentElement).getPropertyValue(varName).trim();
}

function handleBoardClick(e) {
  const k = currentKey();
  if(!localGame && onlineMyRole !== k) return;

  const rect = canvas.getBoundingClientRect();
  const scaleX = canvas.width / rect.width;
  const scaleY = canvas.height / rect.height;
  const mx = (e.clientX - rect.left) * scaleX;
  const my = (e.clientY - rect.top) * scaleY;

  const c = Math.round((mx - CELL_SIZE) / CELL_SIZE);
  const r = Math.round((my - CELL_SIZE) / CELL_SIZE);

  if(r < 0 || r >= BOARD_SIZE || c < 0 || c >= BOARD_SIZE) return;

  // 1. 카드 대상 타겟팅 페이즈 상태일 때 우선 제어 리턴
  if(cardUnderUse) {
    executeBoardCardEffect(r, c);
    return;
  }

  // 2. 착수 단계 진입 상태일 때 가이드라인 위치 업데이트 핸들링
  if(phase === 'place') {
    if(board[r][c] !== 0) return; // 이미 선점된 노드는 불허
    
    // 신규 지점 지정 시 가이드라인 업데이트 후 확정 버튼 풀림 처리
    tempPlacement = { r, c };
    drawBoard();
    
    const placeBtn = document.getElementById(k === 'black' ? 'bPlaceBtn' : 'wPlaceBtn');
    if(placeBtn) placeBtn.disabled = false;
  }
}

function confirmPlace() {
  if(!tempPlacement) return;
  
  const { r, c } = tempPlacement;
  board[r][c] = currentPlayer;
  
  // 오목 승리 검사 트리거 작동
  if(checkWin(r, c, currentPlayer)) {
    drawBoard();
    showWin(currentPlayer);
    if(!localGame) synchronizeOnlineState(true);
    return;
  }

  // 버프 라이프 사이클 감쇄 연산 처리 기믹
  Object.entries(dangerMarks).forEach(([coord, info]) => {
    info.turnsLeft--;
    if(info.turnsLeft <= 0) {
      const [dr, dc] = coord.split(',').map(Number);
      board[dr][dc] = 0;
      delete dangerMarks[coord];
    }
  });

  // 세션 교대 클린업 연산 후 다음 플레이어로 전환
  tempPlacement = null;
  currentPlayer = currentPlayer === 1 ? 2 : 1;
  phase = 'card';
  turnCount++;
  
  dealCards(currentKey(), 1);
  startTimer();
  renderAll();
  updateActionButtons();

  if(!localGame) {
    synchronizeOnlineState();
    setStatus(onlineMyRole === (currentPlayer === 1 ? 'black' : 'white') ? '내 차례입니다.' : '상대방 차례입니다.');
  } else {
    setStatus(`${currentPlayer === 1 ? '흑' : '백'}의 차례입니다.`);
  }
}

/* 오목(5-in-a-row) 정렬 스캔 서치 최적화 유틸 알고리즘 */
function checkWin(r, c, p) {
  const dirs = [[0,1],[1,0],[1,1],[1,-1]];
  for(let i=0; i<dirs.length; i++) {
    let count = 1;
    const [dr, dc] = dirs[i];
    
    // 정방향 스캔
    let nr = r + dr, nc = c + dc;
    while(nr >= 0 && nr < BOARD_SIZE && nc >= 0 && nc < BOARD_SIZE && board[nr][nc] === p) {
      count++; nr += dr; nc += dc;
    }
    // 역방향 스캔
    nr = r - dr; nc = c - dc;
    while(nr >= 0 && nr < BOARD_SIZE && nc >= 0 && nc < BOARD_SIZE && board[nr][nc] === p) {
      count++; nr -= dr; nc -= dc;
    }
    if(count >= 5) return true;
  }
  return false;
}

/* 턴 제한 시간 관리 정밀 타이머 유틸 연산 엔진 */
function startTimer() {
  clearInterval(timerInterval);
  timeLeft = 30;
  const bar = document.getElementById('timerBar');
  const txt = document.getElementById('timerText');
  
  bar.className = 'timer-bar';
  bar.style.width = '100%';
  txt.textContent = timeLeft;

  timerInterval = setInterval(() => {
    timeLeft--;
    txt.textContent = timeLeft;
    
    const pct = (timeLeft / 30) * 100;
    bar.style.width = pct + '%';

    if(timeLeft <= 15) bar.classList.add('warn');
    if(timeLeft <= 7) { bar.classList.remove('warn'); bar.classList.add('danger'); }

    if(timeLeft <= 0) {
      clearInterval(timerInterval);
      handleTimeout();
    }
  }, 1000);
}

function handleTimeout() {
  setStatus("시간 초과! 강제로 턴이 양도 및 정산처리 됩니다.");
  if(phase === 'card') {
    passToPlace();
  } else {
    // 임시 지정 노드 조차 없으면 빈 노드 자동 강제 지정 처리 하이재킹 알고리즘 가동
    if(!tempPlacement) {
      for(let r=0; r<BOARD_SIZE; r++) {
        for(let c=0; c<BOARD_SIZE; c++) {
          if(board[r][c] === 0) { tempPlacement = {r,c}; break; }
        }
        if(tempPlacement) break;
      }
    }
    confirmPlace();
  }
}

/* 승리 종료 조건 및 탈출 시나리오 팝업 인터페이스 */
function showWin(winner) {
  clearInterval(timerInterval);
  const winOverlay = document.getElementById('winOverlay');
  document.getElementById('winTitle').textContent = winner === 1 ? "흑색 진영 승리!" : "백색 진영 승리!";
  document.getElementById('winDesc').textContent = "전략의 승리입니다. 오목 정렬 스택 스캔 매칭이 검증되었습니다.";
  winOverlay.classList.add('show');
}

function afterWin() {
  document.getElementById('winOverlay').classList.remove('show');
  showPage('page-menu');
}

/* Firebase Realtime Database 실시간 네트워크 처리 모듈 */
function createOnlineRoom() {
  if(!db) { alert("Firebase 환경 배정이 구성되지 않았습니다."); return; }
  const code = Math.floor(100000 + Math.random() * 900000).toString();
  onlineRoomId = code;
  onlineMyRole = 'black';

  db.ref('rooms/' + code).set({
    roomCode: code,
    status: 'wait',
    blackNick: currentUser ? currentUser.nickname : '흑',
    whiteNick: '',
    boardState: JSON.stringify(Array(BOARD_SIZE).fill(0).map(() => Array(BOARD_SIZE).fill(0))),
    currentPlayer: 1,
    turnCount: 1,
    phase: 'card',
    lastAction: { type: 'create', by: 'black' }
  }).then(() => {
    document.getElementById('roomCodeText').textContent = code;
    document.getElementById('roomCodeDisplay').style.display = 'flex';
    
    // 상대편 대전 상대 난입 유무 리스너 설치
    db.ref('rooms/' + code + '/status').on('value', snap => {
      if(snap.val() === 'playing') {
        db.ref('rooms/' + code).once('value', roomSnap => {
          const rData = roomSnap.val();
          closeModal('page-online');
          initGame({
            online: true,
            roomId: code,
            myRole: 'black',
            nicknames: { black: rData.blackNick, white: rData.whiteNick }
          });
          listenToOnlineActions();
        });
      }
    });
  });
}

function joinOnlineRoom() {
  if(!db) { alert("Firebase 연동 기능이 활성화되지 않았습니다."); return; }
  const code = document.getElementById('roomCodeInput').value.trim();
  if(code.length !== 6) { alert("방 코드는 숫자 6자리입니다."); return; }

  db.ref('rooms/' + code).once('value', snap => {
    if(!snap.exists()) {
      document.getElementById('joinMsg').textContent = "존재하지 않는 방 코드 번호입니다.";
      document.getElementById('joinMsg').style.display = 'block';
      return;
    }
    const rData = snap.val();
    if(rData.status !== 'wait') {
      alert("이미 게임이 진행 중이거나 마감된 룸 세션 소스입니다.");
      return;
    }

    onlineRoomId = code;
    onlineMyRole = 'white';
    
    db.ref('rooms/' + code).update({
      whiteNick: currentUser ? currentUser.nickname : '백',
      status: 'playing',
      lastAction: { type: 'join', by: 'white' }
    }).then(() => {
      initGame({
        online: true,
        roomId: code,
        myRole: 'white',
        nicknames: { black: rData.blackNick, white: currentUser ? currentUser.nickname : '백' }
      });
      listenToOnlineActions();
    });
  });
}

function synchronizeOnlineState(isWinGame = false) {
  if(!db || !onlineRoomId) return;
  
  db.ref('rooms/' + onlineRoomId).update({
    boardState: JSON.stringify(board),
    currentPlayer: currentPlayer,
    turnCount: turnCount,
    phase: phase,
    status: isWinGame ? 'finished' : 'playing',
    lastAction: {
      type: 'sync',
      by: onlineMyRole,
      timestamp: Date.now()
    }
  });
}

function listenToOnlineActions() {
  if(!db || !onlineRoomId) return;
  
  db.ref('rooms/' + onlineRoomId).on('value', snap => {
    const data = snap.val();
    if(!data) return;
    
    // 상대방 세션 세그먼트에 의해 강제 종료 또는 기권 파괴가 일어났을 때의 에외 처리
    if(data.status === 'wait') return;
    
    if(data.lastAction && data.lastAction.by !== onlineMyRole) {
      board = JSON.parse(data.boardState);
      currentPlayer = data.currentPlayer;
      turnCount = data.turnCount;
      phase = data.phase;
      
      if(data.status === 'finished') {
        const enemyRoleNum = onlineMyRole === 'black' ? 2 : 1;
        showWin(enemyRoleNum);
        return;
      }
      
      renderAll();
      updateActionButtons();
      
      const isMyTurn = onlineMyRole === (currentPlayer === 1 ? 'black' : 'white');
      if(isMyTurn) {
        setStatus("내 차례입니다. 신중하게 전술 카드를 활용하거나 착수하십시오.");
        startTimer();
      } else {
        setStatus("상대방의 전술 대응 턴 대기 중...");
        clearInterval(timerInterval);
      }
    }
  });
}
